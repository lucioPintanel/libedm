143 ulong v;
144 int b;
145 {
146  int i;
147  ulong t = v;
148  for (i = 0; i < b; i++) {
149  if (t & 1L)
150  v |= BITMASK((b - 1) - i);
151  else
152  v &= ~BITMASK((b - 1) - i);
153  t >>= 1;
154  }

178 void cm_nxt(p_cm, ch)
179 p_cm_t p_cm;
180 int ch;
181 {
182  int i;
183  ulong uch = (ulong) ch;
184  ulong topbit = BITMASK(p_cm->cm_width - 1);
185 
186  if (p_cm->cm_refin) uch = reflect(uch, 8);
187  p_cm->cm_reg ^= (uch << (p_cm->cm_width - 8));
188  for (i = 0; i < 8; i++) {
189  if (p_cm->cm_reg & topbit)
190  p_cm->cm_reg = (p_cm->cm_reg << 1) ^ p_cm->cm_poly;
191  else
192  p_cm->cm_reg <<= 1;
193  p_cm->cm_reg &= widmask(p_cm);
194  }
195 }

228 ulong cm_tab(p_cm, index)
229 p_cm_t p_cm;
230 int index;
231 {
232  int lindex;
233  ulong lresult;
234  ulong ltopbit = BITMASK(p_cm->cm_width - 1);
235  ulong linbyte = (ulong) index;
236 
237  if (p_cm->cm_refin) linbyte = reflect(linbyte, 8);
238  lresult = linbyte << (p_cm->cm_width - 8);
239  for (lindex = 0; lindex < 8; lindex++)
240  if (lresult & ltopbit)
241  lresult = (lresult << 1) ^ p_cm->cm_poly;
242  else
243  lresult <<= 1;
244  if (p_cm->cm_refin) lresult = reflect(lresult, p_cm->cm_width);
245 
246  return lresult & widmask(p_cm);
247 }

262 uint8_t cm_width; /* Parameter: Width in bits [8,32]. */
263 ulong cm_poly; /* Parameter: The algorithm's polynomial. */
